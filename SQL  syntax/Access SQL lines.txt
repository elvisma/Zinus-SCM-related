SELECT PORawData.[Plant Desc#], PORawData.PO, PORawData.Material, PORawData.[PO ETA], PORawData.[Final Del Date], PORawData.[Final GR Date], PORawData.[GR Status], PORawData.[PO Qty], PORawData.[GR Qty] AS [Received Qty], [PO Qty]-[GR Qty] AS [In-Tranist Qty], PORawData.[PO Item], ([PO ETA]-(Weekday([PO ETA])-1)) AS WeekofPOETA, ([Final Del Date]-(Weekday([Final Del Date])-1)) AS WeekofFinalDelDate, ([Final GR Date]-(Weekday([Final GR Date])-1)) AS WeekofFinalGRDate, IIf([WeekofFinalGRDate] Is Null,IIf([WeekofFinalDelDate] Is Null,[WeekofPOETA],[WeekofFinalDelDate]),[WeekofFinalGRDate]) AS [New POETA]
FROM PORawData
WHERE (((PORawData.PO) Not Like 'AZD*' And (PORawData.PO) Not Like 'WF*'));



SELECT qry_PO.[Plant Desc#], qry_PO.[New POETA], qry_PO.PO, qry_PO.Material, qry_PO.[PO Qty], qry_PO.[Received Qty], qry_PO.[In-Tranist Qty], qry_PO.[PO Item], qry_PO.[GR Status]
FROM qry_PO;


SELECT qry_PO.[New POETA], qry_PO.Material, qry_PO.PO, qry_PO.[Plant Desc#]
FROM qry_PO
GROUP BY qry_PO.[New POETA], qry_PO.Material, qry_PO.PO, qry_PO.[Plant Desc#];


SELECT SH_Data.Warehouse, SH_Data.[Customer name], DatePart("yyyy",[Order Creation date],1,1) AS [Year#], DatePart("m",[Order Creation date],1,1) AS [Month#], DatePart("ww",[Order Creation date],1,1) AS [Week#], ([Order Creation date]-(Weekday([Order Creation date])-1)) AS WeekofOrderCreation, SH_Data.[Order Creation date], SH_Data.[ZINUS SKU], SH_Data.[QTY#], SH_Data.[Net VAlue]
FROM SH_Data;


SELECT SH_Data.[Customer name], ([Customer Purchase date]-(Weekday([Customer Purchase date])-1)) AS WeekofOrder, SH_Data.[Customer Purchase date], SH_Data.[ZINUS SKU], SH_Data.[QTY#], SH_Data.[DO Qty], SH_Data.Warehouse
FROM SH_Data;


SELECT SHForSetForAll.[Customer name], ([Customer Purchase date]-(Weekday([Customer Purchase date])-1)) AS WeekofOrder, SHForSetForAll.[Customer Purchase date], SHForSetForAll.[ZINUS SKU], SHForSetForAll.[DO Qty]
FROM SHForSetForAll;
